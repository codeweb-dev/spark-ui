---
title: Image Trail
description: A framed, mouse-tracking image trail with momentum, captions, and customizable styling.
category: Signature Components
---

The ImageTrail displays framed media following the user's pointer with physics-based momentum. It supports rounded frame variants, composable captions, custom keyframe animations, distance thresholds, and configurable stacking.

## Properties

<PropsTable>
  <Prop
    name="as"
    type="ElementType"
    default='"div"'
    description="HTML element to render the container as."
  />
  <Prop
    name="threshold"
    type="number"
    default="100"
    description="Distance in pixels the mouse must travel to spawn the next element."
  />
  <Prop
    name="intensity"
    type="number"
    default="0.3"
    description="Momentum factor (0.0001 to 1.0) for tracking lag. Lower is smoother."
  />
  <Prop
    name="keyframes"
    type="DOMKeyframesDefinition"
    default="undefined"
    description="Framer Motion keyframes defining the animation sequence."
  />
  <Prop
    name="keyframesOptions"
    type="AnimationOptions"
    default="undefined"
    description="Framer Motion options for the keyframes animation."
  />
  <Prop
    name="trailElementAnimationKeyframes"
    type="object"
    default="tween/easeOut"
    description="Framer Motion options for the tracking movement."
  />
  <Prop
    name="repeatChildren"
    type="number"
    default="3"
    description="Number of times children are duplicated to prevent trail gaps."
  />
  <Prop
    name="variant"
    type='"plain" | "framed" | "glass"'
    default='"framed"'
    description="Visual treatment applied to ImageTrailItem."
  />
  <Prop
    name="radius"
    type='"none" | "sm" | "md" | "lg" | "xl" | "full"'
    default='"xl"'
    description="Corner radius applied to an ImageTrailItem and its direct image."
  />
  <Prop
    name="baseZIndex"
    type="number"
    default="0"
    description="Starting z-index for the trailing items."
  />
  <Prop
    name="zIndexDirection"
    type='"new-on-top" | "old-on-top"'
    default='"new-on-top"'
    description="Stacking behavior ('new-on-top' or 'old-on-top')."
  />
</PropsTable>
